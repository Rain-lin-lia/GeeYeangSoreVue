父層調用時，在script聲明變量，array[i]就是第i格步驟的內容
例如
<script setup>

const steps = [
  '登入帳號與密碼',
  '填寫個人資料與聯絡方式',
  '確認資訊與同意條款',
  '完成註冊流程'
]
</script>

<template>


  <!-- 步驟區塊 -->
  <div class="container my-4">
    <Step :steps="steps" :activeStep="1">
      <template #default="{ step, index }">
        <div>
          <div><strong>第 {{ index + 1 }} 步</strong></div>
          <div style="font-size: 12px;">{{ step }}</div>
        </div>
      </template>
    </Step>
  </div>


</template>
